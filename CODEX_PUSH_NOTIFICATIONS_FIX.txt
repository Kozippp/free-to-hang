📝 EXAMPLE INTEGRATION:

Here's what good integration looks like in backend/routes/plans.js:

```javascript
const express = require('express');
const router = express.Router();
const verifyToken = require('../middleware/verifyToken');
const supabase = global.supabase;
const { notifyUser, NotificationTemplates } = require('../services/notificationService');

router.post('/', verifyToken, async (req, res) => {
  try {
    const { title, description, date, location, invitedFriends } = req.body;
    const creatorId = req.user.id;

    // Create plan
    const { data: plan, error: planError } = await supabase
      .from('plans')
      .insert({
        title,
        description,
        date,
        location,
        creator_id: creatorId,
        status: 'active'
      })
      .select()
      .single();

    if (planError) throw planError;

    // Add participants
    if (invitedFriends && invitedFriends.length > 0) {
      const participants = invitedFriends.map(friendId => ({
        plan_id: plan.id,
        user_id: friendId,
        status: 'pending'
      }));

      const { error: participantsError } = await supabase
        .from('plan_participants')
        .insert(participants);

      if (participantsError) throw participantsError;

      // 🔔 SEND NOTIFICATIONS TO INVITED USERS
      const { data: creator } = await supabase
        .from('users')
        .select('name')
        .eq('id', creatorId)
        .single();

      for (const friendId of invitedFriends) {
        try {
          const template = NotificationTemplates.plan_invite(
            plan.title,
            creator?.name || 'Someone'
          );
          
          await notifyUser({
            userId: friendId,
            ...template,
            data: { 
              plan_id: plan.id,
              screen: 'PlanDetail'
            },
            triggeredBy: creatorId
          });
          
          console.log(`✅ Notified user ${friendId} about plan invite`);
        } catch (notifError) {
          console.error(`❌ Failed to notify ${friendId}:`, notifError.message);
          // Don't fail the whole request
        }
      }
    }

    res.status(201).json({ plan });
  } catch (error) {
    console.error('Error creating plan:', error);
    res.status(500).json({ error: error.message });
  }
});

module.exports = router;
```
